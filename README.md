# Manual (Linux & Windows)

```bash
# Create the project
composer create-project laravel/laravel directory
# Vue auth
composer require laravel/ui
php artisan ui vue --auth
npm install
npm run dev
# Database
systemctl start mysql
mysql -u antonio -p
# CREATE DATABASE db_name;
php artisan migrate
# Tests
npm install --save-dev jest @vue/test-utils babel-jest babel-core@^7.0.0-0 @babel/preset-env vue-jest
```
